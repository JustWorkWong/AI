# WMS AI Platform V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first shippable version of the WMS AI platform covering `退货质检与处置` and `SOP 辅助执行`, with Vue frontend, .NET 10 backend services, MAF runtime, EF Core persistence, and Kubernetes deployment artifacts generated by Aspire.

**Architecture:** Keep all business truth inside one `wms-domain-service`, isolate `agent-runtime` for AI execution, use `ops-bff` as the page-oriented aggregation layer, and use `Aspire AppHost` as the single source of truth for local orchestration and Kubernetes Helm artifact generation. Kubernetes cluster add-ons such as `ingress-nginx` and `cert-manager` are managed outside the app chart; application workloads and service manifests are generated from Aspire with `aspire publish`.

**Tech Stack:** .NET 10, ASP.NET Core, EF Core, PostgreSQL, pgvector, RabbitMQ, Redis, MinIO, Microsoft Agent Framework, Microsoft.Extensions.AI, Vue 3, TypeScript, Vite, YARP, Aspire, OpenTelemetry, xUnit, Testcontainers, Helm, Kubernetes.

        db.Users.Add(user);

    await db.SaveChangesAsync(ct);

public static class SopChunker
{
    public static IReadOnlyList<(string StepCode, string Text)> Split(string source)
    {
        return source.Split("STEP-", StringSplitOptions.RemoveEmptyEntries)
            .Select(x => $"STEP-{x}")
            .Select(x =>
            {
                var parts = x.Split('|', 2);
                return (parts[0], parts[1]);
            })
            .ToList();
    }
}

public sealed class ReturnDispositionWorkflow

    public async Task<WorkflowResult> RunAsync(ReturnDispositionInput input, RuntimeContext context)
    {
        var returnOrder = await context.InvokeAsync("GetReturnOrderTool", input);
        var sop = await context.InvokeAsync("SearchSopTool", input);
        var cases = await context.InvokeAsync("SearchHistoricalCasesTool", input);

        var suggestion = await context.GenerateAsync("return-disposition", new

            sop,
            cases

        if (suggestion.RequiresApproval)

            var approval = await context.InvokeAsync("RequestDispositionApprovalTool", suggestion);
            await context.CreateCheckpointAsync("approval", approval.ReferenceId);
            return WorkflowResult.WaitingForApproval(approval.ReferenceId);
        }

        await context.InvokeAsync("ApplyDispositionDecisionTool", new
        {
            input.ReturnOrderId,
            suggestion.Outcome,
            input.IdempotencyKey
        });

        return WorkflowResult.Completed();

    public static SopAssistResult Success(object response, object evidence) => new("Success", new { response, evidence });
}

public sealed class SopAssistWorkflow
{
    public async Task<SopAssistResult> RunStepAsync(SopAssistInput input, RuntimeContext context)
    {
        var candidates = await context.InvokeAsync("SearchSopCandidatesTool", input);
        var chunks = await context.InvokeAsync("RetrieveSopChunksTool", new
        {
            input.OperationCode,
            input.StepCode,
            Candidates = candidates
        });

        var evidence = await context.InvokeAsync("RankSopEvidenceTool", chunks);
        if (evidence.Count == 0)
        {
            return SopAssistResult.ManualReviewRequired();
        }

        var response = await context.GenerateAsync("sop-assist", new
        {
            input.OperationCode,
            input.StepCode,
            Evidence = evidence
        });

        await context.CreateCheckpointAsync("step", input.SessionId);
        return SopAssistResult.Success(response, evidence);
    }
}
